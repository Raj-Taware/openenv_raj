"""
Inference script for Quantum Algorithm Optimization Environment.
Stub version — replaced fully in Task 8 with DQN+LLM policy.
"""
import os
import sys
import json
from src.environment import QuantumOptimizationEnv

API_BASE_URL = os.getenv("API_BASE_URL", "https://api.openai.com/v1")
MODEL_NAME = os.getenv("MODEL_NAME", "gpt-4o-mini")
HF_TOKEN = os.getenv("HF_TOKEN", "")


def run_task(task: str) -> float:
    env = QuantumOptimizationEnv(task=task)
    print(f"[START] Task: {task}")
    sys.stdout.flush()
    obs = env.reset()
    done = False
    step_count = 0
    while not done and step_count < env.max_steps:
        action = env.action_space.sample()
        next_obs, reward, done, info = env.step(action)
        log: dict = {
            "step": step_count,
            "action": int(action),
            "reward": round(float(reward), 4),
            "done": bool(done),
        }
        if done and "final_score" in info:
            log["final_score"] = round(float(info["final_score"]), 4)
        print(f"[STEP] {json.dumps(log)}")
        sys.stdout.flush()
        obs = next_obs
        step_count += 1
    final_score = env._get_final_score()
    print(f"[END] Task: {task}, Score: {round(final_score, 4)}")
    sys.stdout.flush()
    return final_score


if __name__ == "__main__":
    tasks = ["parity-optimization", "shors-factoring", "vqe-h2"]
    results: dict = {}
    for task in tasks:
        results[task] = run_task(task)
    print(f"Final Results: {json.dumps(results)}")
    sys.stdout.flush()
```

```python
"""
Quantum Algorithm Optimization Environment.

An OpenEnv-compliant Gymnasium environment where an agent modifies a 3-qubit
quantum circuit to solve one of three tasks by selecting gate-addition actions.

Fidelity is computed via Qiskit Statevector (exact, no shots, instant for 3 qubits).
"""
import gymnasium as gym
import numpy as np
from qiskit import QuantumCircuit
from qiskit.quantum_info import Statevector, SparsePauliOp
from qiskit.circuit.library import QFT
from typing import Dict, Any, Tuple, Optional
from gymnasium import spaces
from .graders import get_grader


# ---------------------------------------------------------------------------
# Pre-compute ideal target statevectors once at module load (< 1 ms each)
# ---------------------------------------------------------------------------

def _build_parity_ideal() -> Statevector:
    """3-qubit parity circuit: CNOT(0→2) then CNOT(1→2) accumulates XOR into qubit 2."""
    circ = QuantumCircuit(3)
    circ.cx(0, 2)
    circ.cx(1, 2)
    return Statevector(circ)


def _build_shors_ideal() -> Statevector:
    """3-qubit QFT (no final swap) — the core subroutine of Shor's algorithm."""
    return Statevector(QFT(3, do_swaps=False))


PARITY_IDEAL_SV: Statevector = _build_parity_ideal()
SHORS_IDEAL_SV: Statevector = _build_shors_ideal()

# 2-qubit H2 Hamiltonian (STO-3G basis, Jordan-Wigner transform).
# Acts on qubits 0 and 1 of the circuit.
H2_HAMILTONIAN = SparsePauliOp.from_list([
    ("II", -0.8105),
    ("ZI",  0.1722),
    ("IZ", -0.2228),
    ("ZZ",  0.1208),
    ("YY", -0.0453),
    ("XX",  0.1722),
])
H2_GROUND_ENERGY = -1.137  # Hartree, FCI/STO-3G


# ---------------------------------------------------------------------------
# Environment
# ---------------------------------------------------------------------------

class QuantumOptimizationEnv(gym.Env):
    """
    OpenEnv for quantum algorithm optimization.

    The agent repeatedly chooses one of 20 discrete gate actions to
    build/modify a 3-qubit quantum circuit.  Each task has a different
    fidelity target.

    Tasks
    -----
    parity-optimization : Achieve high statevector overlap with ideal parity circuit.
    shors-factoring     : Achieve high overlap with 3-qubit QFT.
    vqe-h2              : Minimize H2 molecular energy via variational ansatz.
    """

    metadata = {"render_modes": []}

    def __init__(self, task: str = "parity-optimization", max_steps: int = 50):
        super().__init__()
        assert task in ("parity-optimization", "shors-factoring", "vqe-h2"), \
            f"Unknown task '{task}'. Choose from: parity-optimization, shors-factoring, vqe-h2"

        self.task = task
        self.max_steps = max_steps
        self.current_step: int = 0
        self.qubit_count: int = 3  # All tasks use 3 qubits for uniform action space
        self.circuit: Optional[QuantumCircuit] = None
        self.problem_params: Dict[str, Any] = self._get_problem_params()

        # Fidelity cache — keyed by circuit depth (invalidated on any gate change)
        self._cached_fidelity: Optional[float] = None
        self._cached_depth: int = -1

        # ------------------------------------------------------------------
        # Action space: 20 discrete actions
        #   0–2   : Add H gate to qubit 0, 1, 2
        #   3–8   : Add CNOT (6 ordered qubit-pair combinations of {0,1,2})
        #   9–11  : Add T gate to qubit 0, 1, 2
        #   12–14 : Add S gate to qubit 0, 1, 2
        #   15–17 : Add X gate to qubit 0, 1, 2
        #   18    : Remove last gate (no-op if circuit empty)
        #   19    : No-op
        # ------------------------------------------------------------------
        self.action_space = spaces.Discrete(20)

        self.observation_space = spaces.Dict({
            "circuit_gates": spaces.Sequence(
                spaces.Dict({
                    "name":   spaces.Text(10),
                    "qubits": spaces.Sequence(spaces.Discrete(10)),
                })
            ),
            "qubit_count":      spaces.Discrete(10),
            "current_fidelity": spaces.Box(low=0.0, high=1.0, shape=(1,), dtype=np.float32),
            "problem_params":   spaces.Dict({"value": spaces.Box(low=0.0, high=1.0, shape=(1,))}),
            "steps_remaining":  spaces.Discrete(51),
        })

        self.reset()

    # ------------------------------------------------------------------
    # Core Gymnasium interface
    # ------------------------------------------------------------------

    def reset(
        self,
        *,
        seed: Optional[int] = None,
        options: Optional[Dict[str, Any]] = None,
    ) -> Dict[str, Any]:
        super().reset(seed=seed)
        self.current_step = 0
        self.circuit = QuantumCircuit(self.qubit_count)
        self._invalidate_cache()

        # Seed each task with a minimal starting circuit so the agent
        # begins from a non-trivial state.
        if self.task == "parity-optimization":
            self.circuit.h(0)           # Superposition on first qubit
        elif self.task == "shors-factoring":
            self.circuit.h(0)           # Begin building towards QFT
            self.circuit.h(1)
        elif self.task == "vqe-h2":
            self.circuit.x(0)           # Hartree-Fock reference state |100⟩

        self._invalidate_cache()
        return self._get_observation()

    def step(
        self, action: int
    ) -> Tuple[Dict[str, Any], float, bool, bool, Dict[str, Any]]:
        self.current_step += 1
        info: Dict[str, Any] = {}

        valid = self._apply_action(action)
        penalty = -0.1 if not valid else 0.0

        fidelity = self._compute_fidelity()          # uses cache — called only once here
        reward = penalty + (fidelity - 0.5) * 2.0   # maps [0,1] → [-1, 1]

        terminated = bool(fidelity >= 0.95)
        truncated  = bool(self.current_step >= self.max_steps)
        done       = terminated or truncated

        if done:
            info["final_score"] = self._get_final_score()

        # Legacy 4-tuple compatibility (some callers unpack 4 values)
        observation = self._get_observation()
        return observation, reward, done, info

    # ------------------------------------------------------------------
    # Fidelity (with caching)
    # ------------------------------------------------------------------

    def _invalidate_cache(self) -> None:
        self._cached_fidelity = None
        self._cached_depth = -1

    def _compute_fidelity_uncached(self) -> float:
        """Compute fidelity via Statevector — exact and instant for 3-qubit circuits."""
        try:
            current_sv = Statevector(self.circuit)
        except Exception:
            return 0.0

        if self.task == "parity-optimization":
            overlap = abs(complex(PARITY_IDEAL_SV.inner(current_sv))) ** 2
            return float(np.clip(overlap, 0.0, 1.0))

        elif self.task == "shors-factoring":
            overlap = abs(complex(SHORS_IDEAL_SV.inner(current_sv))) ** 2
            return float(np.clip(overlap, 0.0, 1.0))

        elif self.task == "vqe-h2":
            # Build 2-qubit subcircuit from gates acting only on qubits 0 and 1
            circ_2q = QuantumCircuit(2)
            for instr in self.circuit.data:
                indices = [self.circuit.find_bit(q).index for q in instr.qubits]
                if all(idx < 2 for idx in indices):
                    new_qubits = [circ_2q.qubits[idx] for idx in indices]
                    circ_2q.append(instr.operation, new_qubits)
            try:
                sv_2q = Statevector(circ_2q)
                energy = float(sv_2q.expectation_value(H2_HAMILTONIAN).real)
                score = 1.0 - abs(energy - H2_GROUND_ENERGY) / (abs(H2_GROUND_ENERGY) + 1e-8)
                return float(np.clip(score, 0.0, 1.0))
            except Exception:
                return 0.0

        return 0.5

    def _compute_fidelity(self) -> float:
        """Return fidelity, reusing cached value when circuit depth hasn't changed."""
        depth = len(self.circuit.data)
        if self._cached_fidelity is not None and self._cached_depth == depth:
            return self._cached_fidelity
        self._cached_fidelity = self._compute_fidelity_uncached()
        self._cached_depth = depth
        return self._cached_fidelity

    # ------------------------------------------------------------------
    # Action application
    # ------------------------------------------------------------------

    def _apply_action(self, action: int) -> bool:
        """Modify circuit according to action. Returns True if valid."""
        try:
            if   action == 0:  self.circuit.h(0)
            elif action == 1:  self.circuit.h(1)
            elif action == 2:  self.circuit.h(2)
            elif action == 3:  self.circuit.cx(0, 1)
            elif action == 4:  self.circuit.cx(0, 2)
            elif action == 5:  self.circuit.cx(1, 0)
            elif action == 6:  self.circuit.cx(1, 2)
            elif action == 7:  self.circuit.cx(2, 0)
            elif action == 8:  self.circuit.cx(2, 1)
            elif action == 9:  self.circuit.t(0)
            elif action == 10: self.circuit.t(1)
            elif action == 11: self.circuit.t(2)
            elif action == 12: self.circuit.s(0)
            elif action == 13: self.circuit.s(1)
            elif action == 14: self.circuit.s(2)
            elif action == 15: self.circuit.x(0)
            elif action == 16: self.circuit.x(1)
            elif action == 17: self.circuit.x(2)
            elif action == 18:
                if len(self.circuit.data) > 0:
                    self.circuit.data.pop()
            # action == 19: no-op (intentional pass)
            self._invalidate_cache()
            return True
        except Exception:
            return False

    # ------------------------------------------------------------------
    # Observation and scoring
    # ------------------------------------------------------------------

    def _get_observation(self) -> Dict[str, Any]:
        gates = [
            {
                "name":   instr.operation.name,
                "qubits": [self.circuit.find_bit(q).index for q in instr.qubits],
            }
            for instr in self.circuit.data
        ]
        return {
            "circuit_gates":    gates,
            "qubit_count":      self.qubit_count,
            "current_fidelity": np.array([self._compute_fidelity()], dtype=np.float32),
            "problem_params":   self.problem_params,
            "steps_remaining":  self.max_steps - self.current_step,
        }

    def _get_final_score(self) -> float:
        grader = get_grader(self.task)
        obs = self._get_observation()
        return grader(obs["circuit_gates"], obs["qubit_count"], self.problem_params)

    def _get_problem_params(self) -> Dict[str, Any]:
        if self.task == "parity-optimization":
            return {"parity_bits": [0, 1, 2]}
        elif self.task == "shors-factoring":
            return {"target_number": 15}
        elif self.task == "vqe-h2":
            return {"molecule": "H2", "ground_energy": H2_GROUND_ENERGY}
        return {}

    def state(self) -> Dict[str, Any]:
        """Alias for _get_observation() — OpenEnv spec compatibility."""
        return self._get_observation()
```

"""
Physics-based graders for quantum algorithm tasks.

Each grader accepts (circuit_gates, qubit_count, problem_params) and returns
a float in [0.0, 1.0] based on actual quantum state fidelity or energy.
"""
from typing import Dict, Any, List
import numpy as np
from qiskit import QuantumCircuit
from qiskit.quantum_info import Statevector, SparsePauliOp
from qiskit.circuit.library import QFT

# Fixed-angle gate map used to reconstruct circuits from gate-dict lists
_GATE_MAP = {
    "h":       lambda c, q: c.h(q[0]),
    "cx":      lambda c, q: c.cx(q[0], q[1]),
    "t":       lambda c, q: c.t(q[0]),
    "s":       lambda c, q: c.s(q[0]),
    "x":       lambda c, q: c.x(q[0]),
    "ry":      lambda c, q: c.ry(np.pi / 4, q[0]),
    "rz":      lambda c, q: c.rz(np.pi / 4, q[0]),
    "rx":      lambda c, q: c.rx(np.pi / 4, q[0]),
}

# H2 Hamiltonian (STO-3G, Jordan-Wigner, 2-qubit)
_H2_HAMILTONIAN = SparsePauliOp.from_list([
    ("II", -0.8105),
    ("ZI",  0.1722),
    ("IZ", -0.2228),
    ("ZZ",  0.1208),
    ("YY", -0.0453),
    ("XX",  0.1722),
])
_H2_GROUND_ENERGY = -1.137  # Hartree


def _gates_to_circuit(gates: List[Dict[str, Any]], n_qubits: int) -> QuantumCircuit:
    """Reconstruct a QuantumCircuit from a list of {"name": str, "qubits": List[int]} dicts."""
    circ = QuantumCircuit(n_qubits)
    for gate in gates:
        name = gate.get("name", "")
        qubits = gate.get("qubits", [])
        if name in _GATE_MAP and qubits and all(q < n_qubits for q in qubits):
            try:
                _GATE_MAP[name](circ, qubits)
            except Exception:
                pass  # Skip gates with wrong qubit count
    return circ


def grade_parity_optimization(
    circuit_gates: List[Dict[str, Any]],
    qubit_count: int,
    problem_params: Dict[str, Any],
) -> float:
    """
    Score by statevector overlap with the ideal 3-qubit parity circuit.

    Ideal circuit: CNOT(0→2) then CNOT(1→2).
    This accumulates the XOR of all 3 qubits into qubit 2.
    Score 1.0 = perfect match; 0.0 = orthogonal or empty circuit.
    """
    if not circuit_gates:
        return 0.0
    n = max(qubit_count, 3)
    ideal = QuantumCircuit(n)
    ideal.cx(0, 2)
    ideal.cx(1, 2)
    ideal_sv = Statevector(ideal)
    circ = _gates_to_circuit(circuit_gates, n)
    try:
        overlap = abs(complex(ideal_sv.inner(Statevector(circ)))) ** 2
        return float(np.clip(overlap, 0.0, 1.0))
    except Exception:
        return 0.0


def grade_shors_factoring(
    circuit_gates: List[Dict[str, Any]],
    qubit_count: int,
    problem_params: Dict[str, Any],
) -> float:
    """
    Score by statevector overlap with the 3-qubit QFT (no swap).

    The QFT is the computational core of Shor's period-finding algorithm.
    Score 1.0 = perfect QFT; 0.0 = empty/orthogonal.
    """
    if not circuit_gates:
        return 0.0
    n = max(qubit_count, 3)
    ideal_sv = Statevector(QFT(n, do_swaps=False))
    circ = _gates_to_circuit(circuit_gates, n)
    try:
        overlap = abs(complex(ideal_sv.inner(Statevector(circ)))) ** 2
        return float(np.clip(overlap, 0.0, 1.0))
    except Exception:
        return 0.0


def grade_vqe_h2(
    circuit_gates: List[Dict[str, Any]],
    qubit_count: int,
    problem_params: Dict[str, Any],
) -> float:
    """
    Score by proximity to H2 ground-state energy (-1.137 Ha).

    Builds a 2-qubit subcircuit from gates acting on qubits 0 and 1, computes
    the H2 Hamiltonian expectation value, and maps it to [0, 1].
    Score 1.0 = circuit achieves the exact ground state energy.
    Score 0.0 = circuit achieves zero energy (identity / no-op).
    """
    if not circuit_gates:
        return 0.0
    circ_2q = QuantumCircuit(2)
    for gate in circuit_gates:
        name = gate.get("name", "")
        qubits = [q for q in gate.get("qubits", []) if q < 2]
        if name in _GATE_MAP and qubits:
            try:
                _GATE_MAP[name](circ_2q, qubits)
            except Exception:
                pass
    try:
        sv_2q = Statevector(circ_2q)
        energy = float(sv_2q.expectation_value(_H2_HAMILTONIAN).real)
        score = 1.0 - abs(energy - _H2_GROUND_ENERGY) / (abs(_H2_GROUND_ENERGY) + 1e-8)
        return float(np.clip(score, 0.0, 1.0))
    except Exception:
        return 0.0


def get_grader(task: str):
    """Return the grader function for the given task name."""
    return {
        "parity-optimization": grade_parity_optimization,
        "shors-factoring":     grade_shors_factoring,
        "vqe-h2":              grade_vqe_h2,
    }.get(task, lambda *_: 0.0)
```

```python
"""
PyTorch DQN building blocks for quantum circuit optimization.

Public API
----------
OBS_DIM           : int — size of the encoded observation vector (13).
encode_observation : Dict -> Tensor[OBS_DIM] — converts env obs to feature vector.
DQNNetwork        : nn.Module — maps Tensor[B, OBS_DIM] to Q-values Tensor[B, 20].
ReplayBuffer      : circular replay buffer storing (s, a, r, s', done) transitions.
"""
from collections import deque
from typing import Dict, Any, Tuple
import random

import numpy as np
import torch
import torch.nn as nn

# Gate vocabulary used for histogram encoding.
# Index 9 ("other") catches any gate name not in this list.
_GATE_VOCAB = ["h", "cx", "t", "s", "x", "ry", "rz", "rx", "measure", "other"]

# Observation vector size:
#   10 (gate histogram) + 1 (qubit_count) + 1 (fidelity) + 1 (steps) = 13
OBS_DIM: int = len(_GATE_VOCAB) + 3  # = 13


def encode_observation(obs: Dict[str, Any]) -> torch.Tensor:
    """
    Convert an environment observation dict to a float32 Tensor of shape (OBS_DIM,).

    Feature layout:
        [0–9]  Gate-type histogram, normalized by total gate count.
               Index i = count of _GATE_VOCAB[i] gates / total gates.
               Unknown gate names go into index 9 ("other").
        [10]   qubit_count / 10.0
        [11]   current_fidelity[0]   (already in [0, 1])
        [12]   steps_remaining / 50.0
    """
    gates = obs.get("circuit_gates", [])
    hist = np.zeros(len(_GATE_VOCAB), dtype=np.float32)
    for gate in gates:
        name = gate.get("name", "other")
        idx = _GATE_VOCAB.index(name) if name in _GATE_VOCAB else len(_GATE_VOCAB) - 1
        hist[idx] += 1.0
    total = float(max(1, len(gates)))
    hist /= total

    qubit_norm      = float(obs.get("qubit_count", 3)) / 10.0
    fidelity        = float(obs["current_fidelity"][0])
    steps_norm      = float(obs.get("steps_remaining", 50)) / 50.0

    features = np.concatenate([hist, [qubit_norm, fidelity, steps_norm]])
    return torch.from_numpy(features)  # shape (13,), dtype float32


class DQNNetwork(nn.Module):
    """
    Two-hidden-layer MLP: observation features → Q-values for 20 actions.

    Architecture
    ------------
    Linear(input_dim → hidden_dim) → ReLU
    Linear(hidden_dim → hidden_dim) → ReLU
    Linear(hidden_dim → n_actions)
    """

    def __init__(
        self,
        input_dim: int = OBS_DIM,
        hidden_dim: int = 128,
        n_actions: int = 20,
    ):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, n_actions),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: Tensor of shape (B, input_dim) or (input_dim,)
        Returns:
            Q-values of shape (B, n_actions)
        """
        return self.net(x)


class ReplayBuffer:
    """
    Circular experience replay buffer.

    Stores (state, action, reward, next_state, done) tuples.
    When capacity is exceeded, the oldest transition is overwritten.
    """

    def __init__(self, capacity: int = 10_000):
        self.buffer: deque = deque(maxlen=capacity)

    def push(
        self,
        state: torch.Tensor,
        action: int,
        reward: float,
        next_state: torch.Tensor,
        done: bool,
    ) -> None:
        """Add one transition to the buffer."""
        self.buffer.append((state, int(action), float(reward), next_state, bool(done)))

    def sample(self, batch_size: int) -> Tuple[torch.Tensor, ...]:
        """
        Sample a random batch of transitions.

        Returns
        -------
        states      : Tensor (B, OBS_DIM)
        actions     : Tensor (B,) long
        rewards     : Tensor (B,) float32
        next_states : Tensor (B, OBS_DIM)
        dones       : Tensor (B,) float32  (1.0 = terminal)
        """
        batch = random.sample(self.buffer, batch_size)
        states, actions, rewards, next_states, dones = zip(*batch)
        return (
            torch.stack(states),
            torch.tensor(actions,  dtype=torch.long),
            torch.tensor(rewards,  dtype=torch.float32),
            torch.stack(next_states),
            torch.tensor(dones,    dtype=torch.float32),
        )

    def __len__(self) -> int:
        return len(self.buffer)

"""
DQN training loop for quantum circuit optimization.

Usage (CLI)
-----------
    python -m src.train                        # train all 3 tasks, 500 episodes each
    python -m src.train --task parity-optimization --episodes 200

Each task saves weights to models/dqn_{task_shortname}.pth.
"""
import os
import argparse
import random
from typing import Optional

import torch
import torch.nn.functional as F

from src.environment import QuantumOptimizationEnv
from src.agent import DQNNetwork, ReplayBuffer, encode_observation, OBS_DIM

# ── Hyperparameters ──────────────────────────────────────────────────────────
N_ACTIONS       = 20
HIDDEN_DIM      = 128
LEARNING_RATE   = 1e-3
GAMMA           = 0.99        # discount factor
EPSILON_START   = 1.0         # initial exploration rate
EPSILON_END     = 0.05        # minimum exploration rate
EPSILON_DECAY   = 0.995       # per-episode multiplicative decay
BATCH_SIZE      = 32
REPLAY_CAPACITY = 10_000
TARGET_SYNC_FREQ = 50         # steps between target-net syncs
MIN_BUFFER      = 64          # transitions before training starts


def train_dqn(
    task: str,
    n_episodes: int = 500,
    save_path: Optional[str] = None,
    verbose: bool = True,
) -> DQNNetwork:
    """
    Train a Double DQN policy for the given task.

    Parameters
    ----------
    task       : "parity-optimization" | "shors-factoring" | "vqe-h2"
    n_episodes : number of training episodes
    save_path  : where to save the trained weights (.pth); None = don't save
    verbose    : print progress every 50 episodes

    Returns
    -------
    Trained DQNNetwork in eval() mode.
    """
    env = QuantumOptimizationEnv(task=task)

    policy_net = DQNNetwork(input_dim=OBS_DIM, hidden_dim=HIDDEN_DIM, n_actions=N_ACTIONS)
    target_net = DQNNetwork(input_dim=OBS_DIM, hidden_dim=HIDDEN_DIM, n_actions=N_ACTIONS)
    target_net.load_state_dict(policy_net.state_dict())
    target_net.eval()

    optimizer = torch.optim.Adam(policy_net.parameters(), lr=LEARNING_RATE)
    buffer    = ReplayBuffer(capacity=REPLAY_CAPACITY)

    epsilon    = EPSILON_START
    global_step = 0
    best_score  = 0.0

    for episode in range(n_episodes):
        obs        = env.reset()
        state      = encode_observation(obs)
        ep_reward  = 0.0
        done       = False

        while not done:
            # ε-greedy action selection
            if random.random() < epsilon:
                action = int(env.action_space.sample())
            else:
                policy_net.eval()
                with torch.no_grad():
                    action = int(policy_net(state.unsqueeze(0)).argmax(dim=1).item())

            next_obs, reward, done, info = env.step(action)
            next_state = encode_observation(next_obs)

            buffer.push(state, action, reward, next_state, done)
            state      = next_state
            ep_reward += reward
            global_step += 1

            # ── Training step ────────────────────────────────────────────────
            if len(buffer) >= MIN_BUFFER:
                policy_net.train()
                sb, ab, rb, nsb, db = buffer.sample(BATCH_SIZE)

                # Double DQN: policy_net selects action, target_net evaluates it
                with torch.no_grad():
                    best_next_actions = policy_net(nsb).argmax(dim=1, keepdim=True)
                    next_q = target_net(nsb).gather(1, best_next_actions).squeeze(1)
                    target_q = rb + GAMMA * next_q * (1.0 - db)

                current_q = policy_net(sb).gather(1, ab.unsqueeze(1)).squeeze(1)
                loss = F.smooth_l1_loss(current_q, target_q)

                optimizer.zero_grad()
                loss.backward()
                torch.nn.utils.clip_grad_norm_(policy_net.parameters(), max_norm=10.0)
                optimizer.step()

            # ── Sync target network ──────────────────────────────────────────
            if global_step % TARGET_SYNC_FREQ == 0:
                target_net.load_state_dict(policy_net.state_dict())

        # ── Per-episode bookkeeping ──────────────────────────────────────────
        epsilon = max(EPSILON_END, epsilon * EPSILON_DECAY)
        score   = env._get_final_score()
        if score > best_score:
            best_score = score

        if verbose and (episode + 1) % 50 == 0:
            print(
                f"[{task}] Ep {episode + 1:>4}/{n_episodes} | "
                f"Reward {ep_reward:+6.2f} | "
                f"Best {best_score:.3f} | "
                f"ε {epsilon:.3f}"
            )

    policy_net.eval()

    if save_path:
        os.makedirs(os.path.dirname(save_path) or ".", exist_ok=True)
        torch.save(policy_net.state_dict(), save_path)
        if verbose:
            print(f"[{task}] Saved → {save_path}")

    return policy_net


# ── CLI entry point ──────────────────────────────────────────────────────────

_TASK_PATHS = {
    "parity-optimization": "models/dqn_parity.pth",
    "shors-factoring":     "models/dqn_shors.pth",
    "vqe-h2":              "models/dqn_vqe.pth",
}

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Train DQN for quantum circuit optimization")
    parser.add_argument(
        "--task",
        choices=[*_TASK_PATHS.keys(), "all"],
        default="all",
    )
    parser.add_argument("--episodes", type=int, default=500)
    args = parser.parse_args()

    tasks = list(_TASK_PATHS.keys()) if args.task == "all" else [args.task]
    for t in tasks:
        print(f"\n{'='*60}\nTraining: {t}  ({args.episodes} episodes)\n{'='*60}")
        train_dqn(task=t, n_episodes=args.episodes, save_path=_TASK_PATHS[t])
    print("\nAll models saved to models/")
```

```python
"""
Hybrid policy: PyTorch DQN + OpenAI LLM.

Decision logic
--------------
- If fidelity < LLM_FIDELITY_THRESHOLD AND LLM is available → query LLM for action + reasoning.
  On any LLM failure (network, bad JSON, out-of-range action) → silently fall back to DQN.
- Otherwise → DQN selects the action with the highest Q-value.

The LLM call is via the OpenAI Python client (v1+) and is compatible with any
OpenAI-API-compatible endpoint (set API_BASE_URL env var to override).
"""
import os
import json
from typing import Dict, Any, Tuple, Optional

import numpy as np
import torch

from src.agent import DQNNetwork, encode_observation, OBS_DIM

# LLM is only queried when fidelity is below this threshold
LLM_FIDELITY_THRESHOLD: float = 0.40

_TASK_DESCRIPTIONS: Dict[str, str] = {
    "parity-optimization": (
        "computing the parity (XOR) of 3 qubits. "
        "Ideal circuit: CNOT(qubit 0 → qubit 2), then CNOT(qubit 1 → qubit 2). "
        "This accumulates the XOR of all three qubits into qubit 2."
    ),
    "shors-factoring": (
        "implementing a 3-qubit Quantum Fourier Transform (QFT), "
        "which is the core period-finding subroutine of Shor's algorithm for factoring 15. "
        "Use H gates and T/S phase gates to approximate the QFT."
    ),
    "vqe-h2": (
        "finding the H2 molecular ground-state energy (-1.137 Hartree) using VQE. "
        "Build a variational ansatz on qubits 0 and 1 with rotation (H, T, S) and CNOT gates. "
        "The energy expectation is computed using the STO-3G Jordan-Wigner Hamiltonian."
    ),
}

_ACTION_GUIDE = """\
Actions (choose one integer 0–19):
  0: H on qubit 0     1: H on qubit 1     2: H on qubit 2
  3: CNOT 0→1         4: CNOT 0→2         5: CNOT 1→0
  6: CNOT 1→2         7: CNOT 2→0         8: CNOT 2→1
  9: T on qubit 0    10: T on qubit 1    11: T on qubit 2
 12: S on qubit 0    13: S on qubit 1    14: S on qubit 2
 15: X on qubit 0    16: X on qubit 1    17: X on qubit 2
 18: Remove last gate
 19: No-op"""


class HybridPolicy:
    """
    Selects gate actions using DQN Q-values (always) and LLM reasoning (when fidelity is low).

    Parameters
    ----------
    task       : task name string
    model_path : path to a pretrained .pth weight file; None or missing → random DQN weights
    use_llm    : whether to attempt LLM queries at all
    """

    def __init__(
        self,
        task: str,
        model_path: Optional[str] = None,
        use_llm: bool = True,
    ):
        self.task = task
        self.use_llm = use_llm

        # ── DQN ─────────────────────────────────────────────────────────────
        self.dqn = DQNNetwork(input_dim=OBS_DIM, hidden_dim=128, n_actions=20)
        if model_path and os.path.exists(model_path):
            self.dqn.load_state_dict(torch.load(model_path, map_location="cpu"))
        self.dqn.eval()

        # ── LLM client ──────────────────────────────────────────────────────
        self.llm_client = None
        if use_llm:
            try:
                from openai import OpenAI
                self.llm_client = OpenAI(
                    base_url=os.getenv("API_BASE_URL", "https://api.openai.com/v1"),
                    api_key=os.getenv("HF_TOKEN") or os.getenv("OPENAI_API_KEY") or "dummy",
                )
            except ImportError:
                pass  # openai not installed — will use DQN only

    # ── Public API ───────────────────────────────────────────────────────────

    def select_action(self, observation: Dict[str, Any]) -> Tuple[int, str]:
        """
        Choose an action given the current circuit observation.

        Returns
        -------
        action   : int in [0, 19]
        reasoning: human-readable explanation string
        """
        fidelity = float(observation["current_fidelity"][0])

        should_use_llm = (
            self.use_llm
            and self.llm_client is not None
            and fidelity < LLM_FIDELITY_THRESHOLD
        )

        if should_use_llm:
            try:
                action, reasoning = self._llm_action(observation)
                return action, f"LLM: {reasoning}"
            except Exception as exc:
                reasoning = f"LLM failed ({type(exc).__name__}): {exc}. Falling back to DQN."
                return self._dqn_action(observation), reasoning

        return self._dqn_action(observation), f"DQN policy (fidelity={fidelity:.3f})"

    # ── Private helpers ──────────────────────────────────────────────────────

    def _dqn_action(self, observation: Dict[str, Any]) -> int:
        """Return the action with the highest Q-value."""
        state = encode_observation(observation).unsqueeze(0)
        with torch.no_grad():
            return int(self.dqn(state).argmax(dim=1).item())

    def _llm_action(self, observation: Dict[str, Any]) -> Tuple[int, str]:
        """
        Query the LLM for the next action.

        Returns (action_int, one_sentence_reasoning).
        Raises on any API error, JSON parse failure, or out-of-range action.
        """
        model_name = os.getenv("MODEL_NAME", "gpt-4o-mini")
        depth = len(observation["circuit_gates"])
        recent = observation["circuit_gates"][-5:]
        fidelity = float(observation["current_fidelity"][0])
        steps_left = int(observation.get("steps_remaining", 50))

        prompt = (
            f"You are optimizing a 3-qubit quantum circuit for the task of "
            f"{_TASK_DESCRIPTIONS[self.task]}\n\n"
            f"Circuit depth: {depth} gates\n"
            f"Last 5 gates: {json.dumps(recent)}\n"
            f"Current fidelity: {fidelity:.4f}  (target ≥ 0.95)\n"
            f"Steps remaining: {steps_left}\n\n"
            f"{_ACTION_GUIDE}\n\n"
            "Choose the single best action to improve fidelity toward the target.\n"
            'Respond ONLY with valid JSON (no markdown): '
            '{"action": <int 0-19>, "reasoning": "<one sentence>"}'
        )

        response = self.llm_client.chat.completions.create(
            model=model_name,
            messages=[{"role": "user", "content": prompt}],
            max_tokens=150,
            temperature=0.2,
        )
        raw = response.choices[0].message.content.strip()

        # Strip markdown fences if model wraps response
        if "```" in raw:
            parts = raw.split("```")
            raw = parts[1].lstrip("json").strip() if len(parts) > 1 else raw

        data = json.loads(raw)
        action = int(data["action"])
        reasoning = str(data.get("reasoning", "LLM recommendation"))

        if not 0 <= action <= 19:
            raise ValueError(f"Action {action} out of valid range [0, 19]")

        return action, reasoning
```

#!/usr/bin/env python3
"""
Inference script for Quantum Algorithm Optimization Environment.

Runs a hybrid PyTorch DQN + LLM policy on all 3 tasks and emits structured
logs in the format required by the OpenEnv hackathon.

Required environment variables
-------------------------------
API_BASE_URL  : OpenAI-compatible API base URL (default: https://api.openai.com/v1)
MODEL_NAME    : LLM model name (default: gpt-4o-mini)
HF_TOKEN      : API key / Hugging Face token

Log format (per spec)
---------------------
[START] Task: <task>
[STEP] {"step": N, "action": N, "reasoning": "...", "reward": N.N, "fidelity": N.N, "done": bool}
[END] Task: <task>, Score: N.N
"""
import os
import sys
import json

from src.environment import QuantumOptimizationEnv
from src.policy import HybridPolicy

# ── Required environment variables (hackathon spec) ──────────────────────────
API_BASE_URL = os.getenv("API_BASE_URL", "https://api.openai.com/v1")
MODEL_NAME   = os.getenv("MODEL_NAME", "gpt-4o-mini")
HF_TOKEN     = os.getenv("HF_TOKEN", "")

# Enable LLM when an API key is configured or a custom base URL is set
_USE_LLM: bool = bool(HF_TOKEN) or (API_BASE_URL != "https://api.openai.com/v1")

_MODEL_PATHS = {
    "parity-optimization": "models/dqn_parity.pth",
    "shors-factoring":     "models/dqn_shors.pth",
    "vqe-h2":              "models/dqn_vqe.pth",
}


def run_task(task: str) -> float:
    """
    Run one episode for the given task using the hybrid DQN+LLM policy.

    Prints [START], [STEP]×N, [END] to stdout.
    Returns the final_score in [0.0, 1.0].
    """
    env    = QuantumOptimizationEnv(task=task)
    policy = HybridPolicy(
        task=task,
        model_path=_MODEL_PATHS.get(task),
        use_llm=_USE_LLM,
    )

    print(f"[START] Task: {task}", flush=True)

    obs        = env.reset()
    done       = False
    step_count = 0

    while not done and step_count < env.max_steps:
        action, reasoning = policy.select_action(obs)
        next_obs, reward, done, info = env.step(action)

        log: dict = {
            "step":      step_count,
            "action":    int(action),
            "reasoning": reasoning,
            "reward":    round(float(reward), 4),
            "fidelity":  round(float(next_obs["current_fidelity"][0]), 4),
            "done":      bool(done),
        }
        if done and "final_score" in info:
            log["final_score"] = round(float(info["final_score"]), 4)

        print(f"[STEP] {json.dumps(log)}", flush=True)
        obs        = next_obs
        step_count += 1

    final_score = env._get_final_score()
    print(f"[END] Task: {task}, Score: {round(final_score, 4)}", flush=True)
    return final_score


if __name__ == "__main__":
    tasks   = ["parity-optimization", "shors-factoring", "vqe-h2"]
    results: dict = {}

    for task in tasks:
        results[task] = run_task(task)

    print(f"Final Results: {json.dumps(results)}", flush=True)
